class Node(object):
    """
    构造链表
    """
    def __init__(self, data=None, next=None):
        self.data = data
        self.next = next


def rev(link):
    """
    反转链表
    """
    pre = link
    cur = link.next
    pre.next = None
    while cur:
        tmp = cur.next
        cur.next = pre
        pre = cur
        cur = tmp
    return pre


def traversing(link):
    """
    遍历链表
    """
    temp = link
    while temp:
        print(temp.data)
        temp = temp.next

list = [1, 2, 3, 4, 5, 6, 7, 8, 9]
lin = Node(list[0])
links = lin
for i in list[1:]:
    lin.next = Node(i)
    lin = lin.next

traversing(links) # 输出：[1, 2, 3, 4, 5, 6, 7, 8, 9]
root = rev(links) # 反转链表
traversing(root)  # 输出：[9, 8, 7, 6, 5, 4, 3, 2, 1]
