def binary_search(list, item):
    """
    二分查找：
        输入：[1,3,5,7,9]， 3
        输出： 1
    """
    low = 0
    high = len(list) - 1
    while low <= high:
        mid = (high + low) // 2   
        guess = list[mid]
        if guess > item:
            high = mid - 1
        elif guess < item:
            low = mid + 1
        else:
            return mid
    return None
mylist = [1,3,5,7,9]
ans = binary_search(mylist, 3)
print(ans)
